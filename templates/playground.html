Základní matematika

b = 5
print(f"Součet: {a + b}")

Seznamy a cykly
# Práce se seznamy
barvy = ["červená", "modrá", "zelená"]
print("Barvy:")
for barva in barvy:
    print(f"- {barva}")

# Přidání nové barvy
barvy.append("žlutá")
print(f"Celkem barev: {len(barvy)}")
